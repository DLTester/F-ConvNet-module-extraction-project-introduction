    @return info: dict åŒ…å«æ‰€æœ‰ä¸­é—´ç»“æœï¼Œä¾¿äº debug/å¯è§†åŒ–
    ------------------------------------------------------------------
    """

    # ------------------------------------------------------------------
    # 1) 2D Detection
    # ------------------------------------------------------------------
    bbox_2d = self.detector_2d(image)      # (B,4)

    # ------------------------------------------------------------------
    # 2) ï¼ˆå¯é€‰ï¼‰LiDAR backbone
    # ------------------------------------------------------------------
    lidar_feat = None
    if self.lidar_backbone is not None:
        lidar_feat = self.lidar_backbone(pointcloud)

    # ------------------------------------------------------------------
    # 3) ç‚¹äº‘æŠ•å½± â†’ å›¾åƒå¹³é¢åæ ‡ (uv) å’Œæ·±åº¦ (z)
    # ------------------------------------------------------------------
    uv, depth = torch.ops.frustum_ops.project_points_to_image(pointcloud, K)

    # ------------------------------------------------------------------
    # 4) æ ¹æ® 2D bbox è¿‡æ»¤ç‚¹
    # ------------------------------------------------------------------
    box_mask = torch.ops.frustum_ops.filter_points_in_box2d(
        uv,
        bbox_2d,
        image.shape[-1],   # width
        image.shape[-2],   # height
        True
    )

    # ------------------------------------------------------------------
    # 5) æ·±åº¦åˆ‡åˆ†ï¼ˆSliding Frustumï¼‰
    # ------------------------------------------------------------------
    bins, frustum_id = torch.ops.frustum_ops.build_depth_bins_and_assign(
        depth,
        self.cfg["u"],
        self.cfg["s"],
        self.cfg["z_min"],
        self.cfg["z_max"],
    )

    # ------------------------------------------------------------------
    # 6) å°†ç‚¹æŒ‰ frustum åˆ†ç»„
    # ------------------------------------------------------------------
    frustum_points, frustum_mask = torch.ops.frustum_ops.group_points_by_frustum(
        pointcloud,
        frustum_id,
        self.cfg["T"],     # total number of frustums
        self.cfg["M"],     # max points per frustum
    )

    # ------------------------------------------------------------------
    # 7) PointNet ç¼–ç ï¼ˆå¯¹æ¯ä¸ª frustum çš„ç‚¹ç‰¹å¾ï¼‰
    # ------------------------------------------------------------------
    norm_p = torch.ops.frustum_ops.normalize_frustum_points_with_centroid(
        frustum_points, frustum_mask
    )

    point_feats = self.point_mlp(norm_p)

    frustum_feats = torch.ops.frustum_ops.max_pool_frustum_features(
        point_feats, frustum_mask
    )

    # ------------------------------------------------------------------
    # 8) æ„å»ºåºåˆ—ç‰¹å¾å›¾ï¼ˆä¸º FCN åšå‡†å¤‡ï¼‰
    # ------------------------------------------------------------------
    seq_map = torch.ops.frustum_ops.build_conv_feature_map(frustum_feats)

    # ------------------------------------------------------------------
    # 9) F-ConvNet FCN ç‰¹å¾æå–
    # ------------------------------------------------------------------
    x1 = self.block1(seq_map)
    x2 = self.block2(x1)
    x3 = self.block3(x2)

    xx1 = self.up1(x1)
    xx2 = self.up2(x2)
    xx3 = self.up3(x3)

    # å¤šå°ºåº¦èåˆç®—å­ï¼ˆè‡ªå®šä¹‰ C++/Python ç®—å­ï¼‰
    fused = torch.ops.frustum_ops.align_and_concat_multiscale(xx1, xx2, xx3)

    # é€šé“å‹ç¼©
    fcn_out = self.reduce(fused)

    # ------------------------------------------------------------------
    # è¿”å›æ‰€æœ‰ä¸­é—´ç»“æœï¼ˆç”¨äº Debug / å¯è§†åŒ–ï¼‰
    # ------------------------------------------------------------------
    info = {
        "bbox_2d": bbox_2d,
        "uv": uv,
        "depth": depth,
        "frustum_points": frustum_points,
        "frustum_mask": frustum_mask,
        "point_feats": point_feats,
        "frustum_feats": frustum_feats,
        "seq_map": seq_map,
        "xx1": xx1,
        "xx2": xx2,
        "xx3": xx3,
        "fused": fused,
        "lidar_feat": lidar_feat,
    }

    return fcn_out, info
